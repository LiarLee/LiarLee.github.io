# pyautogui说明  
pyautogui.position()  #  获取鼠标位置  
pyautogui.locateOnScreen()  #  对屏幕截图，获取图片文件所对应的屏幕坐标  
pyautogui.click()   #  模拟鼠标点击  
pyautogui.doubleclick()  #  模拟鼠标双击  
pyautogui.moveTo()  #  移动到屏幕坐标位置  
pyautogui.moveRel()     #  移动固定的坐标距离  
pyautogui.dragRel()     #  按住鼠标拖拽  

# 代码如下：  
